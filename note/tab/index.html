新手学习python，经常要使用python命令行查找一些不熟悉的使用方法等等，但是python命令行下没有自带tab补全的功能,这里也是百度一下记录笔记

vi /usr/local/lib/python2.7/dist-packages/tab.py
#!/usr/bin/env python 
# python startup file 
import sys
import readline
import rlcompleter
import atexit
import os
# tab completion 
readline.parse_and_bind('tab: complete')
# history file 
histfile = os.path.join(os.environ['HOME'], '.pythonhistory')
try:
    readline.read_history_file(histfile)
except IOError:
    pass
atexit.register(readline.write_history_file, histfile)
del os, histfile, readline, rlcompleter
